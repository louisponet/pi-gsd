    2. Create a Command subclass per subcommand.

    4. The run() method calls the same lib function with the same arguments.

    Example pattern for verify:
    ```typescript
    import { Command, Flags, Args } from "@oclif/core";
    import { runVerify } from "../lib/verify.js";

    export class VerifyCommand extends Command {
      static description = "Run verification for a phase";
      static flags = {
        phase: Flags.string({ description: "Phase number", char: "p" }),
        all: Flags.boolean({ description: "Verify all phases" }),
        raw: Flags.boolean({ description: "Raw output" }),
      };
      async run() {
        const { flags } = await this.parse(VerifyCommand);
        await runVerify({ phase: flags.phase, all: flags.all, raw: flags.raw });
      }
    }
    ```
  </action>
