<interfaces>
oclif non-plugin routing pattern for single CJS bundle:

Because oclif's run() expects to be invoked as a plugin CLI (it scans OCLIF_COMMANDS env or a
commands directory), we use oclif's lower-level parse() approach instead:

```typescript
// src/cli.ts — oclif-based router
import { CommandClass } from "./commands/index.js";

const COMMAND_MAP: Record<string, typeof Command> = {
  "state json": StateJsonCommand,
  "state update": StateUpdateCommand,
  "state reset": StateResetCommand,
  "init plan-phase": InitPlanPhaseCommand,
  "roadmap analyze": RoadmapAnalyzeCommand,
  "roadmap get-phase": RoadmapGetPhaseCommand,
  "config-get": ConfigGetCommand,
  "config-set": ConfigSetCommand,
  // ... all commands
  "wxp process": WxpProcessCommand,
};

// Router: join first N tokens to match command key, then delegate
const argv = process.argv.slice(2);
const key = findCommandKey(argv, COMMAND_MAP);
if (key) {
  const Cmd = COMMAND_MAP[key];
  const remainingArgs = argv.slice(key.split(" ").length);
  await Cmd.run(remainingArgs);
} else {
  printHelp(); // auto-generated from COMMAND_MAP
}
```

    ```typescript
    import { Command, Flags } from "@oclif/core";
    import { readFileSync } from "node:fs";
    import { join, dirname } from "node:path";
    import { processWxp } from "../wxp/index.js";
    import { DEFAULT_SHELL_ALLOWLIST } from "../wxp/security.js";
    import type { WxpSecurityConfig } from "../wxp/schema.js";

    export class WxpProcessCommand extends Command {
      static description = "Process a workflow file or inline content through the WXP engine";
      static flags = {
        input: Flags.string({ description: "Inline WXP content to process" }),
        file: Flags.string({ description: "Path to workflow file to process", char: "f" }),
        cwd: Flags.string({ description: "Project working directory" }),
      };

      async run() {
        const { flags } = await this.parse(WxpProcessCommand);
        const cwd = flags.cwd ?? process.cwd();
        const extDir = typeof __filename !== "undefined" ? dirname(__filename) : "";
        const pkgHarness = join(extDir, "..", ".gsd", "harnesses", "pi", "get-shit-done");

        const security: WxpSecurityConfig = {
          trustedPaths: [pkgHarness, join(cwd, ".pi", "gsd")],
          shellAllowlist: [...DEFAULT_SHELL_ALLOWLIST],
          shellTimeoutMs: 30_000,
        };

        let content: string;
        let filePath: string;

        if (flags.file) {
          content = readFileSync(flags.file, "utf8");
          filePath = flags.file;
        } else if (flags.input) {
          content = flags.input;
          filePath = join(pkgHarness, "_inline.md"); // trusted path for inline
        } else {
          this.error("Provide --input or --file");
        }

        const result = processWxp(content, filePath, { security });
        this.log(result);
      }
    }
    ```

    ```typescript
    function findCommandKey(
      argv: string[],
      map: Record<string, unknown>,
    ): string | undefined {
      // Try 2-token key first, then 1-token
      const twoToken = argv.slice(0, 2).join(" ");
      const oneToken = argv[0] ?? "";
      return map[twoToken] ? twoToken : map[oneToken] ? oneToken : undefined;
    }
    ```
