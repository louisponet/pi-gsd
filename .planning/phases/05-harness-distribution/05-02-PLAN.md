    ```typescript
    // Version comparison: check for outdatable harness files
    try {
      const pkgJsonPath = join(dirname(extFile), "..", "..", "package.json");
      const pkgJson = existsSync(pkgJsonPath)
        ? (JSON.parse(readFileSync(pkgJsonPath, "utf8")) as { version?: string })
        : null;
      const pkgVersion = pkgJson?.version ?? "0.0.0";

      function semverGt(a: string, b: string): boolean {
        const pa = a.split(".").map(Number);
        const pb = b.split(".").map(Number);
        for (let i = 0; i < 3; i++) {
          if ((pa[i] ?? 0) > (pb[i] ?? 0)) return true;
          if ((pa[i] ?? 0) < (pb[i] ?? 0)) return false;
        }
        return false;
      }

      function readVersionTag(filePath: string): { v: string; doNotUpdate: boolean } | null {
        try {
          const content = readFileSync(filePath, "utf8");
          const m = /<gsd-version\s+v="([^"]+)"(\s+do-not-update)?\s*\/>/.exec(content);
          if (!m) return null;
          return { v: m[1], doNotUpdate: !!m[2] };
        } catch { return null; }
      }

      function findOutdatedFiles(harnessSrc: string, harnessDest: string, pkgVer: string): string[] {
        const outdated: string[] = [];
        try {
          const files = readdirSync(harnessDest, { withFileTypes: true });
          for (const f of files) {
            if (f.isDirectory()) {
              outdated.push(...findOutdatedFiles(
                join(harnessSrc, f.name),
                join(harnessDest, f.name),
                pkgVer,
              ));
            } else if (f.name.endsWith(".md")) {
              const destPath = join(harnessDest, f.name);
              const tag = readVersionTag(destPath);
              if (!tag || tag.doNotUpdate) continue;
              if (semverGt(pkgVer, tag.v)) {
                outdated.push(f.name);
              }
            }
          }
        } catch { /* silent */ }
        return outdated;
      }

      const outdatedFiles = findOutdatedFiles(pkgHarness, projectHarness, pkgVersion);
      if (outdatedFiles.length > 0) {
        ctx.ui.notify(
          `GSD harness update available (v${pkgVersion}).\n` +
          `${outdatedFiles.length} workflow file(s) have newer versions available.\n` +
          `Run: pi-gsd-tools harness update`,
          "info",
        );
      }
    } catch {
      /* silent — version check is advisory only */
    }
    ```

    Create `src/commands/harness.ts`:

    ```typescript
    import { Command, Args } from "@oclif/core";
    import { copyFileSync, existsSync, mkdirSync, readFileSync, readdirSync, writeFileSync } from "node:fs";
    import { dirname, join } from "node:path";
    import { readVersionTag } from "../wxp/index.js";

    export class HarnessUpdateCommand extends Command {
      static description = "Update project harness files from the package harness";
      static args = {
        action: Args.string({
          description: "Update action: y (overwrite all), n (list only), pick (select files), diff (create diffs)",
          options: ["y", "n", "pick", "diff"],
          default: "n",
        }),
      };

      async run() {
        const { args } = await this.parse(HarnessUpdateCommand);
        const extDir = typeof __filename !== "undefined" ? dirname(__filename) : "";
        const pkgRoot = join(extDir, "..");
        const pkgHarness = join(pkgRoot, ".gsd", "harnesses", "pi", "get-shit-done");
        const projectHarness = join(process.cwd(), ".pi", "gsd");
        const pkgVersion = (JSON.parse(readFileSync(join(pkgRoot, "package.json"), "utf8")) as { version: string }).version;

        function semverGt(a: string, b: string): boolean {
          const pa = a.split(".").map(Number);
          const pb = b.split(".").map(Number);
          for (let i = 0; i < 3; i++) {
            if ((pa[i] ?? 0) > (pb[i] ?? 0)) return true;
            if ((pa[i] ?? 0) < (pb[i] ?? 0)) return false;
          }
          return false;
        }

        // Find all outdated files
        const outdated: Array<{ rel: string; srcPath: string; destPath: string; currentV: string }> = [];
        function scanDir(src: string, dest: string, relBase: string) {
          try {
            for (const entry of readdirSync(src, { withFileTypes: true })) {
              const rel = relBase ? `${relBase}/${entry.name}` : entry.name;
              if (entry.isDirectory()) {
                scanDir(join(src, entry.name), join(dest, entry.name), rel);
              } else if (entry.name.endsWith(".md")) {
                const destPath = join(dest, entry.name);
                const tag = existsSync(destPath) ? readVersionTag(destPath) : null;
                if (!tag || tag.doNotUpdate) continue;
                if (semverGt(pkgVersion, tag.v)) {
                  outdated.push({ rel, srcPath: join(src, entry.name), destPath, currentV: tag.v });
                }
              }
            }
          } catch { /* skip inaccessible dirs */ }
        }
        scanDir(pkgHarness, projectHarness, "");

        if (outdated.length === 0) {
          this.log("All harness files are up to date.");
          return;
        }

        this.log(`${outdated.length} file(s) have updates available (package v${pkgVersion}):`);
        outdated.forEach((f, i) => this.log(`  ${i + 1}. ${f.rel} (currently v${f.currentV})`));

        switch (args.action) {
          case "y":
            for (const f of outdated) {
              mkdirSync(dirname(f.destPath), { recursive: true });
              copyFileSync(f.srcPath, f.destPath);
              this.log(`Updated: ${f.rel}`);
            }
            this.log(`Done. ${outdated.length} file(s) updated.`);
            break;
          case "n":
            this.log("No changes made. To update, run: pi-gsd-tools harness update y");
            break;
          case "pick":
            this.log("Enter file numbers to update (comma-separated), or 'all':");
            // In non-interactive context, print instructions and exit
            this.log("Example: pi-gsd-tools harness update y  (to update all)");
            break;
          case "diff":
            for (const f of outdated) {
              const srcContent = readFileSync(f.srcPath, "utf8");
              const diffPath = `${f.destPath}.gsd-update.diff`;
              writeFileSync(diffPath, `--- current (v${f.currentV})\n+++ package (v${pkgVersion})\n\n${srcContent}`);
              this.log(`Diff written: ${diffPath}`);
            }
            break;
        }
      }
    }
    ```
