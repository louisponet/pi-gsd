    4. Add `copyHarness` function before the default export:
       ```typescript
       function copyHarness(
         src: string,
         dest: string,
       ): { filesCopied: number; symlinksReplaced: number } {
         let filesCopied = 0;
         let symlinksReplaced = 0;
         try {
           mkdirSync(dest, { recursive: true });
           const entries = readdirSync(src, { withFileTypes: true });
           for (const entry of entries) {
             const srcPath = join(src, entry.name);
             const destPath = join(dest, entry.name);
             if (entry.isDirectory()) {
               const sub = copyHarness(srcPath, destPath);
               filesCopied += sub.filesCopied;
               symlinksReplaced += sub.symlinksReplaced;
             } else {
               // Check if dest is a symlink
               try {
                 const stat = lstatSync(destPath);
                 if (stat.isSymbolicLink()) {
                   // Replace symlink with real copy
                   const { unlinkSync } = await import("node:fs");
                   unlinkSync(destPath);
                   copyFileSync(srcPath, destPath);
                   symlinksReplaced++;
                   continue;
                 }
                 // Real file exists — skip (never overwrite)
                 continue;
               } catch {
                 // File doesn't exist — copy it
               }
               copyFileSync(srcPath, destPath);
               filesCopied++;
             }
           }
         } catch {
           /* silent — never block session startup */
         }
         return { filesCopied, symlinksReplaced };
       }
       ```

       Note: Replace the `await import` with synchronous `unlinkSync` import at the top of the file.
