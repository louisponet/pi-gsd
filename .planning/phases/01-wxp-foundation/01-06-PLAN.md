  <action>
    Create `src/wxp/__tests__/integration.test.ts`:

    ```typescript
    import { vi, describe, it, expect, beforeEach } from "vitest";

    vi.mock("node:child_process", () => ({
      execFileSync: vi.fn().mockReturnValue("mocked-shell-output"),
    }));
    vi.mock("node:fs", async (importOriginal) => {
      const actual = await importOriginal<typeof import("node:fs")>();
      return {
        ...actual,
        readFileSync: vi.fn((p: string, enc: string) => {
          if (p.includes("included.md")) return "<gsd-paste name=\"inner\" />";
          return actual.readFileSync(p, enc as BufferEncoding);
        }),
        existsSync: vi.fn().mockReturnValue(true),
      };
    });

    import { processWxp, WxpProcessingError } from "../index.js";
    import type { WxpSecurityConfig } from "../schema.js";

    const TRUSTED = "/trusted/harness";
    const security: WxpSecurityConfig = {
      trustedPaths: [TRUSTED],
      shellAllowlist: ["pi-gsd-tools", "git", "node", "cat", "ls", "echo", "find"],
      shellTimeoutMs: 30_000,
    };
    const cfg = { security };

    describe("processWxp() integration", () => {
      it("replaces <gsd-paste> with shell output (basic shell output)", () => {
        const content = `<gsd-execute><shell command="pi-gsd-tools" result="r">state json</shell></gsd-execute>\n<gsd-paste name="r" />`;
        const result = processWxp(content, `${TRUSTED}/test.md`, cfg);
        expect(result).toContain("mocked-shell-output");
        expect(result).not.toMatch(/<gsd-paste/);
      });

      it("preserves <gsd-paste> inside code fences (code-fence skip)", () => {
        const content = `\`\`\`\n<gsd-paste name="x" />\n\`\`\``;
        const result = processWxp(content, `${TRUSTED}/test.md`, cfg);
        expect(result).toContain("<gsd-paste");
      });

      it("executes children of <if> only when condition matches", () => {
        const content = `<gsd-execute><shell command="echo" result="flag">yes</shell><if var="flag"><equals value="yes"/><shell command="echo" result="branch">branch-ran</shell></if></gsd-execute><gsd-paste name="branch" />`;
        const result = processWxp(content, `${TRUSTED}/test.md`, cfg);
        expect(result).toContain("mocked-shell-output");
      });

      it("throws WxpProcessingError for untrusted file path", () => {
        expect(() =>
          processWxp("<gsd-paste name=\"x\" />", "/untrusted/.planning/evil.md", cfg)
        ).toThrow(WxpProcessingError);
      });

      it("throws WxpProcessingError for undefined paste variable", () => {
        const content = `<gsd-paste name="undefined_var" />`;
        expect(() =>
          processWxp(content, `${TRUSTED}/test.md`, cfg)
        ).toThrow(WxpProcessingError);
      });

      it("TST-03: output contains zero <gsd- substrings outside code fences after processing", () => {
        const content = `<gsd-execute><shell command="pi-gsd-tools" result="r">state json</shell></gsd-execute>\n<gsd-paste name="r" />\nSome text\n\`\`\`\n<gsd-paste name="preserved" />\n\`\`\``;
        const result = processWxp(content, `${TRUSTED}/test.md`, cfg);
        // Strip code fences from result before checking
        const withoutFences = result.replace(/```[\s\S]*?```/g, "");
        expect(withoutFences).not.toMatch(/<gsd-/);
      });
    });
    ```
