<interfaces>
Two-pass argument parsing (WXP-02):
  Pass 1: scan $ARGUMENTS string for --flag (boolean) and --flag value (string) patterns.
         Remove them from the token list.
  Pass 2: remaining tokens are positionals, assigned left-to-right.
         The LAST positional defined in <gsd-arguments> is greedy: it consumes all
         remaining tokens joined by spaces.
  Result stored in variables under each arg's name.

Condition evaluation (WXP-04):
  <if var="X"><equals value="Y"> → evaluate vars.get("X") === "Y"
  <if var="X"><starts-with value="Y"> → evaluate vars.get("X").startsWith("Y")
  Children execute only when condition is true; skipped otherwise.

String-op split (WXP-05):
  splits vars.get(node.var) by node.delimiter
  result stored as newline-joined string under node.result variable name

Paste (WXP-06):
  Replace <gsd-paste name="X" /> in document text with vars.get("X").
  If vars.get("X") is undefined → throw WxpPasteError immediately (abort, no partial output).
</interfaces>
</context>

    - Pass 2: assign remaining tokens to positionals left-to-right. Last positional is greedy.
    - All parsed values are stored in the variable store under each arg's name.
  </behavior>
  <action>
    Create `src/wxp/arguments.ts`:

    ```typescript
    import type { ArgumentsNode } from "./schema.js";
    import type { VariableStore } from "./variables.js";

    export function parseArguments(
      node: ArgumentsNode,
      rawArguments: string,
      vars: VariableStore,
    ): void {
      const tokens = rawArguments.trim().split(/\s+/).filter(Boolean);
      const consumed = new Set<number>();

      // Pass 1: extract flags
      for (const flagDef of node.flags) {
        const flagIndex = tokens.indexOf(`--${flagDef.name}`);
        if (flagIndex === -1) {
          vars.set(flagDef.name, flagDef.boolean ? "false" : "");
          continue;
        }
        consumed.add(flagIndex);
        if (flagDef.boolean) {
          vars.set(flagDef.name, "true");
        } else {
          const valueIndex = flagIndex + 1;
          if (valueIndex < tokens.length && !tokens[valueIndex].startsWith("--")) {
            vars.set(flagDef.name, tokens[valueIndex]);
            consumed.add(valueIndex);
          } else {
            vars.set(flagDef.name, "");
          }
        }
      }

      // Pass 2: remaining tokens → positionals
      const remaining = tokens.filter((_, i) => !consumed.has(i));
      node.positionals.forEach((pos, idx) => {
        if (idx === node.positionals.length - 1 && pos.greedy) {
          vars.set(pos.name, remaining.slice(idx).join(" "));
        } else {
          vars.set(pos.name, remaining[idx] ?? "");
        }
      });
    }
    ```

    ```typescript
    import { extractCodeFenceRegions } from "./parser.js";
    import type { VariableStore } from "./variables.js";

    export class WxpPasteError extends Error {
      constructor(
        public readonly variableName: string,
        public readonly variableSnapshot: Record<string, string>,
      ) {
        super(`<gsd-paste name="${variableName}" /> references undefined variable '${variableName}'`);
        this.name = "WxpPasteError";
      }
    }

    export function applyPaste(content: string, vars: VariableStore): string {
      const deadZones = extractCodeFenceRegions(content);
      const pasteRegex = /<gsd-paste\s+name="([^"]+)"\s*\/>/g;
      let result = content;
      let offset = 0;

      const matches: Array<{ index: number; full: string; name: string }> = [];
      let m: RegExpExecArray | null;
      while ((m = pasteRegex.exec(content)) !== null) {
        const idx = m.index;
        const inDead = deadZones.some(([s, e]) => idx >= s && idx < e);
        if (!inDead) matches.push({ index: idx, full: m[0], name: m[1] });
      }

      for (const match of matches) {
        const value = vars.get(match.name);
        if (value === undefined) {
          throw new WxpPasteError(match.name, vars.snapshot());
        }
        const adjustedIdx = match.index + offset;
        result =
          result.slice(0, adjustedIdx) +
          value +
          result.slice(adjustedIdx + match.full.length);
        offset += value.length - match.full.length;
      }

      return result;
    }
    ```
