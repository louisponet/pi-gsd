<interfaces>
Code-fence skip rule: any text between opening ``` and closing ``` (on their own lines)
is treated as opaque - WXP tags inside are NOT parsed.

Parser input: raw file content (string) + file path
Parser output: WxpDocument (filePath + operations array)

The parser uses regex/string scanning, NOT a full XML parser - WXP tags are well-formed
but the surrounding document is markdown, not XML.

Shell node execution:
  execFileSync(command, args, { encoding: "utf8", timeout: 30_000, windowsHide: true })
  stdout is trimmed and stored in the named variable.
  Any non-zero exit or timeout throws a WxpShellError.

<gsd-include> extensions:
  INC-01: <gsd-include path="..." include-arguments /> - appends $ARGUMENTS text after include
  INC-02: <gsd-include path="..."><gsd-arguments><arg name="x" as="y"/></gsd-arguments></gsd-include>
          maps caller variable x into included file's namespace as y
  INC-03: Variable collision from included files is resolved with owner-prefix (handled in variables.ts)
</interfaces>

    import { WxpDocumentSchema } from "./schema.js";
    import type { WxpDocument, WxpOperation, IncludeNode } from "./schema.js";

    /** Returns [start, end] pairs of code-fence regions (indices in content). */
    export function extractCodeFenceRegions(content: string): Array<[number, number]> {
      const regions: Array<[number, number]> = [];
      const fenceRegex = /^```[^\n]*\n[\s\S]*?^```/gm;
      let m: RegExpExecArray | null;
      while ((m = fenceRegex.exec(content)) !== null) {
        regions.push([m.index, m.index + m[0].length]);
      }
      return regions;
    }

    /** Returns true if position is inside any code-fence region. */
    function inDeadZone(pos: number, regions: Array<[number, number]>): boolean {
      return regions.some(([s, e]) => pos >= s && pos < e);
    }

    /**
     * Parse raw markdown content into a WxpDocument AST.
     * Tags inside fenced code blocks are skipped (WXP-01).
     */
    export function parseWxpDocument(content: string, filePath: string): WxpDocument {
      const deadZones = extractCodeFenceRegions(content);
      const operations: WxpOperation[] = [];

      // Match <gsd-execute>...</gsd-execute> blocks
      const executeRegex = /<gsd-execute>([\s\S]*?)<\/gsd-execute>/g;
      let m: RegExpExecArray | null;
      while ((m = executeRegex.exec(content)) !== null) {
        if (inDeadZone(m.index, deadZones)) continue;
        const inner = m[1];
        operations.push({ type: "execute", children: parseExecuteChildren(inner) });
      }

      // Match <gsd-paste name="..." />
      const pasteRegex = /<gsd-paste\s+name="([^"]+)"\s*\/>/g;
      while ((m = pasteRegex.exec(content)) !== null) {
        if (inDeadZone(m.index, deadZones)) continue;
        operations.push({ type: "paste", name: m[1] });
      }

      // Match <gsd-arguments>...</gsd-arguments>
      const argsRegex = /<gsd-arguments>([\s\S]*?)<\/gsd-arguments>/g;
      while ((m = argsRegex.exec(content)) !== null) {
        if (inDeadZone(m.index, deadZones)) continue;
        operations.push(parseArgumentsBlock(m[1]));
      }

      // Match <gsd-include> tags (INC-01, INC-02)
      const includeRegex = /<gsd-include\s([^>]*?)(?:\/>|>([\s\S]*?)<\/gsd-include>)/g;
      while ((m = includeRegex.exec(content)) !== null) {
        if (inDeadZone(m.index, deadZones)) continue;
        operations.push(parseIncludeTag(m[1], m[2] ?? ""));
      }

      // Match <gsd-version v="..." (do-not-update)? />
      const versionRegex = /<gsd-version\s+v="([^"]+)"(\s+do-not-update)?\s*\/>/g;
      while ((m = versionRegex.exec(content)) !== null) {
        if (inDeadZone(m.index, deadZones)) continue;
        operations.push({ type: "version", v: m[1], doNotUpdate: !!m[2] });
      }

      return WxpDocumentSchema.parse({ filePath, operations });
    }

    function parseExecuteChildren(inner: string): WxpOperation[] {
      // parse <shell>, <if>, <string-op> children
      const children: WxpOperation[] = [];

      const shellRegex = /<shell\s+command="([^"]+)"(?:\s+result="([^"]+)")?([\s\S]*?)<\/shell>/g;
      let m: RegExpExecArray | null;
      while ((m = shellRegex.exec(inner)) !== null) {
        const argsText = m[3] ?? "";
        const args = argsText.trim().split(/\s+/).filter(Boolean);
        children.push({ type: "shell", command: m[1], result: m[2] ?? "result", args });
      }

      const ifRegex = /<if\s+var="([^"]+)">([\s\S]*?)<\/if>/g;
      while ((m = ifRegex.exec(inner)) !== null) {
        const ifVar = m[1];
        const ifInner = m[2];
        const equalsM = /<equals\s+value="([^"]+)"\s*\/>/.exec(ifInner);
        const startsM = /<starts-with\s+value="([^"]+)"\s*\/>/.exec(ifInner);
        const cond = equalsM
          ? { type: "equals" as const, value: equalsM[1] }
          : startsM ? { type: "starts-with" as const, value: startsM[1] } : null;
        if (!cond) continue;
        const ifChildren = parseExecuteChildren(ifInner);
        children.push({ type: "if", var: ifVar, condition: cond, children: ifChildren });
      }

      const stringOpRegex = /<string-op\s+op="([^"]+)"\s+var="([^"]+)"\s+delimiter="([^"]+)"\s+result="([^"]+)"\s*\/>/g;
      while ((m = stringOpRegex.exec(inner)) !== null) {
        children.push({ type: "string-op", op: "split", var: m[2], delimiter: m[3], result: m[4] });
      }

      return children;
    }

    function parseArgumentsBlock(inner: string): WxpOperation {
      const positionals: Array<{ name: string; greedy: boolean }> = [];
      const flags: Array<{ name: string; boolean: boolean }> = [];
      const posRegex = /<positional\s+name="([^"]+)"(\s+greedy)?\s*\/>/g;
      const flagRegex = /<flag\s+name="([^"]+)"(\s+boolean)?\s*\/>/g;
      let m: RegExpExecArray | null;
      while ((m = posRegex.exec(inner)) !== null) positionals.push({ name: m[1], greedy: !!m[2] });
      while ((m = flagRegex.exec(inner)) !== null) flags.push({ name: m[1], boolean: !!m[2] });
      return { type: "arguments", positionals, flags };
    }

    function parseIncludeTag(attrs: string, children: string): IncludeNode {
      const pathM = /path="([^"]+)"/.exec(attrs);
      const inclArgs = /include-arguments/.test(attrs);
      const argMappings: Array<{ name: string; as: string }> = [];
      const argMapRegex = /<arg\s+name="([^"]+)"\s+as="([^"]+)"\s*\/>/g;
      let m: RegExpExecArray | null;
      while ((m = argMapRegex.exec(children)) !== null) {
        argMappings.push({ name: m[1], as: m[2] });
      }
      return {
        type: "include",
        path: pathM?.[1] ?? "",
        includeArguments: inclArgs,
        argMappings,
      };
    }
    ```

    import { execFileSync } from "node:child_process";
    import { checkAllowlist } from "./security.js";
    import type { ShellNode, WxpSecurityConfig } from "./schema.js";
    import type { VariableStore } from "./variables.js";

    export class WxpShellError extends Error {
      constructor(
        public readonly node: ShellNode,
        public readonly stderr: string,
        public readonly variableSnapshot: Record<string, string>,
        message: string,
      ) {
        super(message);
        this.name = "WxpShellError";
      }
    }

    /**
     * Execute a <shell> node using execFileSync.
     * Stores stdout in vars under node.result.
     * Throws WxpShellError on allowlist violation, non-zero exit, or timeout.
     */
    export function executeShell(
      node: ShellNode,
      vars: VariableStore,
      config: WxpSecurityConfig,
    ): void {
      const check = checkAllowlist(node.command, config);
      if (!check.ok) {
        throw new WxpShellError(node, "", vars.snapshot(), check.reason);
      }

      // Interpolate variable references in args: ${varname} → value
      const resolvedArgs = node.args.map((arg) =>
        arg.replace(/\$\{([^}]+)\}/g, (_, name: string) => vars.get(name) ?? ""),
      );

      let stdout = "";
      let stderr = "";
      try {
        stdout = execFileSync(node.command, resolvedArgs, {
          encoding: "utf8",
          timeout: config.shellTimeoutMs,
          windowsHide: true,
        }).trim();
      } catch (err) {
        const e = err as { stderr?: string; message?: string };
        stderr = e.stderr ?? e.message ?? String(err);
        throw new WxpShellError(
          node,
          stderr,
          vars.snapshot(),
          `Shell command '${node.command}' failed: ${stderr}`,
        );
      }

      vars.set(node.result, stdout);
    }
    ```
