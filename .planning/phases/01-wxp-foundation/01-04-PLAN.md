<interfaces>
Resolution loop (WXP-08):
  Iteration order per cycle:
    1. Process all <gsd-include> nodes (inline file content)
    2. Process all <gsd-arguments> blocks (parse $ARGUMENTS into vars)
    3. Process all <gsd-execute> blocks (run shells, conditions, string-ops)
    4. Apply all <gsd-paste> replacements to document text
    5. Re-parse the updated text (new tags may have been injected by paste)
    6. Repeat until no unprocessed tags remain or 50 iterations hit

Error notification format (WXP-09):
  ```
  WXP Processing Error
  File: <path>
  Error: <message>
  Variable Namespace: { "var1": "value1", ... }
  Pending Operations: [list of unresolved tag types]
  Completed Operations: [list of resolved tag types]
  ```
  This must be emitted via ctx.ui.notify (in the extension) OR thrown as WxpProcessingError
  so the caller (extension) can display it.
</interfaces>

    ```typescript
    import { evaluateCondition } from "./conditions.js";
    import { executeShell, WxpShellError } from "./shell.js";
    import { executeStringOp } from "./string-ops.js";
    import type { ExecuteBlock, WxpSecurityConfig, WxpOperation } from "./schema.js";
    import type { VariableStore } from "./variables.js";

    export class WxpExecutionError extends Error {
      constructor(
        public readonly cause: Error,
        public readonly variableSnapshot: Record<string, string>,
        message: string,
      ) {
        super(message);
        this.name = "WxpExecutionError";
      }
    }

    function executeOperation(
      op: WxpOperation,
      vars: VariableStore,
      config: WxpSecurityConfig,
    ): void {
      switch (op.type) {
        case "shell":
          executeShell(op, vars, config);
          break;
        case "if":
          if (evaluateCondition(op, vars)) {
            for (const child of op.children) {
              executeOperation(child, vars, config);
            }
          }
          break;
        case "string-op":
          executeStringOp(op, vars);
          break;
        case "execute":
          executeBlock(op, vars, config);
          break;
        default:
          // paste, arguments, include handled by index.ts resolution loop
          break;
      }
    }

    export function executeBlock(
      block: ExecuteBlock,
      vars: VariableStore,
      config: WxpSecurityConfig,
    ): void {
      try {
        for (const child of block.children) {
          executeOperation(child, vars, config);
        }
      } catch (err) {
        if (err instanceof WxpShellError || err instanceof Error) {
          throw new WxpExecutionError(
            err,
            vars.snapshot(),
            `Execute block failed: ${err.message}`,
          );
        }
        throw err;
      }
    }
    ```

    - Then runs the resolution loop (max 50 iterations):
        1. Re-parse the current text to get the document AST.
        2. If no operations remain → done, return current text.
        3. Process <gsd-include> nodes (read and inline file content).
        4. Process <gsd-arguments> node if present.

        6. Apply <gsd-paste> replacements.
        7. Loop.

    - After loop completion (or before returning): strip any remaining <gsd-*> tags outside code fences (TST-03 requirement - asserted in tests).

  <action>

    ```typescript
    import { readFileSync } from "node:fs";
    import path from "node:path";
    import { parseArguments } from "./arguments.js";
    import { evaluateCondition } from "./conditions.js";
    import { executeBlock, WxpExecutionError } from "./executor.js";
    import { applyPaste, WxpPasteError } from "./paste.js";
    import { parseWxpDocument, extractCodeFenceRegions } from "./parser.js";
    import { checkTrustedPath, DEFAULT_SHELL_ALLOWLIST } from "./security.js";
    import { createVariableStore } from "./variables.js";
    import { WxpSecurityConfigSchema } from "./schema.js";
    import type { WxpSecurityConfig, IncludeNode, ArgumentsNode, ExecuteBlock } from "./schema.js";

    export const MAX_ITERATIONS = 50;

    export class WxpProcessingError extends Error {
      constructor(
        public readonly filePath: string,
        public readonly causeError: Error,
        public readonly variableSnapshot: Record<string, string>,
        public readonly pendingOps: string[],
        public readonly completedOps: string[],
      ) {
        super(
          [
            `WXP Processing Error`,
            `File: ${filePath}`,
            `Error: ${causeError.message}`,
            `Variable Namespace: ${JSON.stringify(variableSnapshot, null, 2)}`,
            `Pending Operations: [${pendingOps.join(", ")}]`,
            `Completed Operations: [${completedOps.join(", ")}]`,
          ].join("\n"),
        );
        this.name = "WxpProcessingError";
      }
    }

    export interface WxpConfig {
      security: WxpSecurityConfig;
      rawArguments?: string;
    }

    export function processWxp(
      content: string,
      filePath: string,
      config: WxpConfig,
    ): string {
      const vars = createVariableStore();
      const completedOps: string[] = [];
      let currentText = content;
      let iterations = 0;

      // Security: validate trusted path
      const pathCheck = checkTrustedPath(filePath, config.security);
      if (!pathCheck.ok) {
        throw new WxpProcessingError(filePath, new Error(pathCheck.reason), {}, [], []);
      }

      try {
        while (iterations < MAX_ITERATIONS) {
          iterations++;
          const doc = parseWxpDocument(currentText, filePath);

          const includes = doc.operations.filter((op): op is IncludeNode => op.type === "include");
          const argsNodes = doc.operations.filter((op): op is ArgumentsNode => op.type === "arguments");
          const execBlocks = doc.operations.filter((op): op is ExecuteBlock => op.type === "execute");
          const hasPaste = currentText.match(/<gsd-paste\s+name="[^"]+"\s*\/>/);

          if (includes.length === 0 && argsNodes.length === 0 && execBlocks.length === 0 && !hasPaste) {
            break; // nothing left to process
          }

          // Step 1: resolve includes
          for (const inc of includes) {
            const incPath = path.resolve(path.dirname(filePath), inc.path);
            const incCheck = checkTrustedPath(incPath, config.security);
            if (!incCheck.ok) throw new Error(incCheck.reason);
            let incContent = readFileSync(incPath, "utf8");
            if (inc.includeArguments && config.rawArguments) {
              incContent += `\n${config.rawArguments}`;
            }
            currentText = currentText.replace(
              new RegExp(`<gsd-include[^>]*?${escapeRegex(inc.path)}[^>]*?(?:\\/>|>[\\s\\S]*?<\\/gsd-include>)`),
              incContent,
            );
            completedOps.push("include");
          }

          // Step 2: arguments
          for (const argsNode of argsNodes) {
            parseArguments(argsNode, config.rawArguments ?? "", vars);
            completedOps.push("arguments");
          }

          // Step 3: execute blocks
          for (const block of execBlocks) {
            executeBlock(block, vars, config.security);
            completedOps.push("execute");
          }

          // Step 4: paste
          if (hasPaste) {
            currentText = applyPaste(currentText, vars);
            completedOps.push("paste");
          }
        }

        if (iterations >= MAX_ITERATIONS) {
          throw new Error(`WXP resolution loop exceeded ${MAX_ITERATIONS} iterations - possible circular include`);
        }

        // Strip any remaining <gsd-*> tags outside code fences
        currentText = stripRemainingWxpTags(currentText);
        return currentText;
      } catch (err) {
        const e = err instanceof Error ? err : new Error(String(err));
        const doc = parseWxpDocument(currentText, filePath);
        const pending = doc.operations.map((op) => op.type);
        throw new WxpProcessingError(filePath, e, vars.snapshot(), pending, completedOps);
      }
    }

    function stripRemainingWxpTags(content: string): string {
      const deadZones = extractCodeFenceRegions(content);
      // Strip <gsd-execute>...</gsd-execute>, <gsd-paste .../>, <gsd-arguments>...</gsd-arguments>
      // Only outside dead zones
      return content.replace(
        /(<gsd-(?:execute|arguments|include|version)[^>]*?>[\s\S]*?<\/gsd-(?:execute|arguments|include|version)>|<gsd-(?:paste|version)\s[^>]*?\/>)/g,
        (match, _g, offset: number) => {
          const inDead = deadZones.some(([s, e]) => offset >= s && offset < e);
          return inDead ? match : "";
        },
      );
    }

    function escapeRegex(str: string): string {
      return str.replace(/[.*+?^${}()|[\]\\]/g, "\\$&");
    }

    // Re-export key types for extension consumers
    export type { WxpSecurityConfig };
    export { DEFAULT_SHELL_ALLOWLIST, WxpSecurityConfigSchema };
    ```
